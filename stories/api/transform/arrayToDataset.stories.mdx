## arrayToDataset(普通对象数组转为dataset)
```tsx

import {
    arrayToDataset,
} from '@jd/auto-chart/lib/transform';

const source = [
    {
        name: "维度1",
        value: 1,
        value2: 2
    },
    {
        name: "维度2",
        value: 1,
        value2: 2
    },
    {
        name: "维度3",
        value: 1,
        value2: 2
    }
];
const alias = {
    name: "name",
    value: "攻击力",
    value2: "耐久力"
};

const newDataset = arrayToDataset(source, alias));

```
